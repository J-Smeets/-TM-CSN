  SSH setup

	Router(config)#hostname R1_CHI
	R1_CHI(config)#no ip domain lookup
	R1_CHI(config)#ip domain-name SafeUS
	
	R1_CHI(config)#line con 0
	R1_CHI(config-line)#password cisco
	R1_CHI(config-line)#login
	R1_CHI(config)#crypto key generate rsa general-keys modulus 1024
	R1_CHI(config)#line vty 0 15
	R1_CHI(config-line)#transport input ssh	
	R1_CHI(config)#ip ssh version 2
	R1_CHI(config)#username cisco secret cisco
