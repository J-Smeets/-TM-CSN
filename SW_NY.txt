SSH Setup

	Switch(config)#host Sw_NY
	Sw_NY(config)#ip domain-name SafeUS
	Sw_NY(config)#crypto key generate rsa general-keys modulus 1024
	Sw_NY(config)#line vty 0 15
	Sw_NY(config-line)#transport input ssh	
	Sw_NY(config)#ip ssh version 2
	Sw_NY(config)#username cisco secret cisco

  Vlans

	Switch(config)#vlan 10
	Switch(config-vlan)#name Data
	Switch(config-vlan)#int vlan 10
	Switch(config-if)#no shut

	Switch(config-if)#vlan 20
	Switch(config-vlan)#name Wireless
	Switch(config-vlan)#int vlan 20
	Switch(config-if)#no shut

	Switch(config-if)#vlan 100

	Switch(config-vlan)#int vlan 100

	Switch(config-if)#vlan 200

  Switchport interfaces

	Sw_NY(config-if)#int g0/1

	Sw_NY(config-if)#switchport trunk allowed vlan 10,20,100,200,99
	Sw_NY(config-if)#switchport trunk native vlan 99

	Sw_NY(config-if)#int g0/2
